# Ergogen config for split ergonomic keyboard with Cherry MX ULP switches
# Layout: 3x5 + 3 thumb keys per half (36 keys total)
# Key spacing: 18mm (Cherry ULP keycap ~15mm + ~3mm gap)

units:
  kx: 18
  ky: 18
  px: kx + 1
  py: ky + 1

points:
  mirror:
    ref: matrix_inner_home
    distance: 22
  zones:
    matrix:
      anchor:
        shift: [100, -100]
      columns:
        outer:
          key:
            column_net: C0
            stagger: 0
            spread: kx
        pinky:
          key:
            column_net: C1
            stagger: 2
            spread: kx
        ring:
          key:
            column_net: C2
            stagger: 5
            spread: kx
        middle:
          key:
            column_net: C3
            stagger: 2.5
            spread: kx
        inner:
          key:
            column_net: C4
            stagger: -2.5
            spread: kx
      rows:
        bottom:
          row_net: R2
        home:
          row_net: R1
        top:
          row_net: R0
    thumb:
      anchor:
        ref: matrix_inner_bottom
        shift: [-4, -20]
      columns:
        inner:
          key:
            column_net: C3
            splay: -10
            spread: kx
        middle:
          key:
            column_net: C4
            splay: -10
            spread: kx
        outer:
          key:
            column_net: C5
            splay: -10
            spread: kx
      rows:
        cluster:
          row_net: R3

outlines:
  _keycaps:
    - what: rectangle
      where: true
      size: [kx - 0.5, ky - 0.5]
  board:
    - what: rectangle
      where: true
      size: [px, py]
      bound: true
      fillet: 2
  switch_cutouts:
    - what: rectangle
      where: true
      size: [13.8, 13.8]
      bound: false

cases:
  bottom_plate:
    - what: outline
      name: board
      extrude: 1.2
  switch_plate:
    - what: outline
      name: board
      extrude: 1.5
    - what: outline
      name: switch_cutouts
      extrude: 1.5
      operation: subtract

pcbs:
  keyboard:
    outlines:
      main:
        outline: board
        layer: Edge.Cuts
    footprints:
      choc_hotswap:
        what: choc
        where: true
        params:
          from: "{{column_net}}"
          to: "{{colrow}}"
          keycaps: true
          reverse: false
          hotswap: true
      diode:
        what: diode
        where: true
        params:
          from: "{{colrow}}"
          to: "{{row_net}}"
        adjust:
          shift: [0, 5]
